    \item Exercise $2.3$ \\
    \item Exercise $2.4$ \\
    \item Exercise $2.5$ \\
    \item Exercise $2.10$ \\
  \end{enumerate}
\end{prob}

\mathbf{e},\mathbf{e},\mathbf{e}), X=W=Y=Z=I$. Thus $\Delta(\mathbf{x},\mathbf{w},

  -A^{-1} & X & O & A^{-1} \\

  -A & I & O & O \\
  O & O & A^T & I \\

  \end{bmatrix}$
\end{proof}
